Learn by Example

Every concept is paired with clear, runnable code.

// Closure — a function that remembers its outer scope
function makeCounter() {
  let count = 0;             // private variable

  return {
    increment: () => ++count,
    decrement: () => --count,
    value:     () => count,
  };
}

const counter = makeCounter();
counter.increment();           // 1
counter.increment();           // 2
console.log(counter.value()); // 2
// Promises — handle async work cleanly
function fetchUser(id) {
  return fetch(`/api/users/${id}`)
    .then(res => res.json())
    .catch(err => console.error(err));
}

// Or with async/await (cleaner!)
async function getUser(id) {
  try {
    const res  = await fetch(`/api/users/${id}`);
    const data = await res.json();
    return data;
  } catch (err) {
    console.error(err);
  }
}
// Destructuring — unpack values elegantly
const user = { name: 'Aman', age: 22, role: 'dev' };

// Object destructuring
const { name, age } = user;

// With rename + default
const { role: title = 'guest' } = user;

// Array destructuring
const [first, , third] = [10, 20, 30];

// In function params
function greet({ name, role }) {
  return `Hi ${name}, you are a ${role}!`;
}
console.log(greet(user)); // Hi Aman, you are a dev!
